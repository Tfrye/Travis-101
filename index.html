export default function Travis101() {

  const [screen, setScreen] = useState(0);

  const TOTAL = 9;

  const touchRef = useRef({ startX:0, startY:0, startTime:0, locked:null });

  const goTo = useCallback(s => { if (s>=0&&s<TOTAL) setScreen(s); }, []);


  useEffect(() => {

    const h = e => {

      if (e.key==="ArrowRight"||e.key==="ArrowDown") goTo(screen+1);

      if (e.key==="ArrowLeft"||e.key==="ArrowUp") goTo(screen-1);

    };

    window.addEventListener("keydown",h);

    return ()=>window.removeEventListener("keydown",h);

  }, [screen, goTo]);


  const onTouchStart = useCallback(e => {

    const t=e.touches[0];

    touchRef.current={ startX:t.clientX, startY:t.clientY, startTime:Date.now(), locked:null };

  },[]);

  const onTouchMove = useCallback(e => {

    const t=e.touches[0];

    const { startX, startY, locked }=touchRef.current;

    const dx=t.clientX-startX, dy=t.clientY-startY;

    if (locked===null) touchRef.current.locked=Math.abs(dx)>Math.abs(dy)?"h":"v";

    if (touchRef.current.locked==="h") e.preventDefault();

  },[]);

  const onTouchEnd = useCallback(e => {

    const t=e.changedTouches[0];

    const { startX, startTime, locked }=touchRef.current;

    const dx=t.clientX-startX, dt=Date.now()-startTime;

    if (locked==="h"&&Math.abs(dx)>40&&dt<500){ if(dx<0)goTo(screen+1); else goTo(screen-1); }

  },[screen,goTo]);


  return (

    <div onTouchStart={onTouchStart} onTouchMove={onTouchMove} onTouchEnd={onTouchEnd}

      style={{ width:"100%", height:"100vh", position:"relative", overflow:"hidden",

        background:P.navy, fontFamily:"'DM Sans',sans-serif", touchAction:"none", userSelect:"none" }}>

      <style>{`

        @import url('https://fonts.googleapis.com/css2?family=Playfair+Display:ital,wght@0,300;0,400;1,300;1,400&family=DM+Sans:ital,wght@0,400;0,500;0,600;0,700;1,400&display=swap');

        *{box-sizing:border-box;margin:0;padding:0} button{font-family:inherit}

        ::-webkit-scrollbar{width:3px} ::-webkit-scrollbar-track{background:transparent}

        ::-webkit-scrollbar-thumb{background:${P.cream}22;border-radius:3px}

        @keyframes floatBounce{0%,100%{transform:translateY(0) scale(1)}33%{transform:translateY(-12px) scale(1.03)}66%{transform:translateY(6px) scale(0.97)}}

        @keyframes sparkle{0%,100%{opacity:0;transform:scale(0) rotate(0deg)}50%{opacity:1;transform:scale(1) rotate(180deg)}}

        @keyframes slideUpBounce{0%{opacity:0;transform:translateY(60px) scale(0.9)}60%{opacity:1;transform:translateY(-8px) scale(1.02)}100%{opacity:1;transform:translateY(0) scale(1)}}

        @keyframes pulse{0%,100%{box-shadow:0 4px 30px ${P.blueLight}44}50%{box-shadow:0 4px 40px ${P.blueLight}88}}

        @keyframes popIn{0%{opacity:0;transform:scale(0.9) translateY(10px)}60%{opacity:1;transform:scale(1.02) translateY(-2px)}100%{opacity:1;transform:scale(1) translateY(0)}}

        @keyframes wiggle{0%,100%{transform:rotate(0deg)}25%{transform:rotate(-3deg)}75%{transform:rotate(3deg)}}

        @keyframes fadeIn{from{opacity:0}to{opacity:1}}

      `}</style>

      <CoverScreen      isActive={screen===0} onStart={()=>goTo(1)}/>

      <KolbeScreen      isActive={screen===1}/>

      <DiscScreen       isActive={screen===2}/>

      <BlindSpotsScreen isActive={screen===3}/>

      <ConnectorScreen  isActive={screen===4}/>

      <UserManualScreen isActive={screen===5}/>

      <DecoderScreen    isActive={screen===6}/>

      <CheatSheetScreen isActive={screen===7}/>

      <CloseScreen      isActive={screen===8} onRestart={()=>goTo(0)}/>

      <div style={{ position:"absolute", bottom:0, left:0, right:0, zIndex:30,

        background:`linear-gradient(transparent,${P.navy}cc 40%)`, paddingTop:20, paddingBottom:8 }}>

        {screen>0&&screen<TOTAL-1&&(

          <div style={{ fontFamily:"'DM Sans',sans-serif", fontSize:11, color:`${P.cream}44`,

            textAlign:"center", letterSpacing:"0.1em", marginBottom:4 }}>← swipe to navigate →</div>

        )}

        <ProgressDots current={screen} total={TOTAL} onNav={goTo}/>

      </div>

    </div>

  );

}
